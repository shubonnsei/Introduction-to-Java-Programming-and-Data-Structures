 27  
 28        // Create a Scanner for the file

 30  
 31        // Read text from the file
 32        while (input.hasNext()) {
 33          System.out.println(input.nextLine());
 34        }
 35  
 36        // Close the file
 37        input.close();
 38      }
 39      else {
 40        System.out.println("No file selected");
 41      }
 42    }
