  3  public class PrintPyramid {
  4    public static void main(String[] args) {

  7  
  8      // Prompt the user to enter the number of lines

 11

 15      }
 16

 19        // Print NUMBER_OF_LINES – row) leading spaces

 21          System.out.print("   ");
 22

 24        for (int num = row; num >= 1; num--)
 25          System.out.print((num >= 10) ? " " + num : "  " + num);
 26  
 27        // Print ending numbers 2, 3, ..., row – 1, row
 28        for (int num = 2; num <= row; num++)
 29          System.out.print((num >= 10) ? " " + num : "  " + num);
 30  
 31        // Start a new line
 32        System.out.println();
 33      }
 34    }
