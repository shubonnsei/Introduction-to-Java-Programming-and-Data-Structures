 29  class MyArcs extends Pane {
 30    private void paint() {  
 31      double centerX = getWidth() / 2;
 32      double centerY = getHeight() / 2;
 33      double radiusX = 0.8 * getWidth() / 2;
 34      double radiusY = 0.8 * getHeight() / 2;
 35      
 36      Arc arc1 = new Arc(centerX, centerY, radiusX, radiusY, 30, 35); 
 37      arc1.setFill(Color.RED); // Set fill color
 38      arc1.setType(ArcType.ROUND); // Set arc type
 39      
 40      Arc arc2 = new Arc(centerX, 
 41        centerY, radiusX, radiusY, 30 + 90, 35);
 42      arc2.setFill(Color.WHITE);
 43      arc2.setType(ArcType.OPEN);
 44      arc2.setStroke(Color.BLACK);
 45  
 46      Arc arc3 = new Arc(centerX, 
 47        centerY, radiusX, radiusY, 30 + 180, 35);
 48      arc3.setFill(Color.WHITE);
 49      arc3.setType(ArcType.CHORD);
 50      arc3.setStroke(Color.BLACK);
 51      getChildren().add(new Text(20, 170, "arc3: chord")); 
 52      
 53      Arc arc4 = new Arc(centerX, 
 54        centerY, radiusX, radiusY, 30 + 270, 35);
 55      arc4.setFill(Color.GREEN);
 56      arc4.setType(ArcType.CHORD);
 57      arc4.setStroke(Color.BLACK);
 58      getChildren().clear();
 59   
 60      getChildren().addAll(arc1, arc2, arc3, arc4, 
 61        new Text(centerX + radiusX - 100, 
 62          centerY - radiusY, "arc1: round"),
 63        new Text(centerX - radiusX, centerY - radiusY, "arc2: open"),
 64        new Text(centerX - radiusX, centerY + radiusY, "arc3: chord"),
 65        new Text(centerX + radiusX - 100,
 66          centerY + radiusY, "arc4: chord")); 
 67    }
 68    
 69    @Override
 70    public void setWidth(double width) {
 71      super.setWidth(width);
 72      paint();
 73    }
 74    
 75    @Override
 76    public void setHeight(double height) {
 77      super.setHeight(height);
 78      paint();
 79    }
 80  }
