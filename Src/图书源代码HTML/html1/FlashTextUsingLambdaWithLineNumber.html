1  import javafx.application.Application;
  2  import javafx.application.Platform;
  3  import javafx.scene.Scene;
  4  import javafx.scene.control.Label;
  5  import javafx.scene.layout.StackPane;
  6  import javafx.stage.Stage;
  7  
  8  public class FlashTextUsingLambda extends Application {
  9    private String text = "";
 10    
 11    @Override // Override the start method in the Application class
 12    public void start(Stage primaryStage) {   
 13      StackPane pane = new StackPane();
 14      Label lblText = new Label("Programming is fun");
 15      pane.getChildren().add(lblText);
 16      
 17      new Thread( () -> {
 18        try {
 19          while (true) {
 20            if (lblText.getText().trim().length() == 0)
 21              text = "Welcome";
 22            else
 23              text = "";
 24  
 25            Platform.runLater(() -> lblText.setText(text));
 26  
 27            Thread.sleep(200);
 28          }
 29        }
 30        catch (InterruptedException ex) {
 31        }
 32      }).start();
 33      
 34      // Create a scene and place it in the stage
 35      Scene scene = new Scene(pane, 200, 50);
 36      primaryStage.setTitle("FlashText"); // Set the stage title
 37      primaryStage.setScene(scene); // Place the scene in the stage
 38      primaryStage.show(); // Display the stage
 39    }
 40  
 41    /**
 42     * The main method is only needed for the IDE with limited
 43     * JavaFX support. Not needed for running from the command line.
 44     */
 45    public static void main(String[] args) {
 46      launch(args);
 47    }
 48  }
