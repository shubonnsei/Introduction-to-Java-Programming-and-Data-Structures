  5  public class CountOccurrenceOfWordsStream {
  6    public static void main(String[] args) {
  7      // Set text in a string

  9        + "Have a good visit. Have fun!";

 13          Collectors.groupingBy(String::toLowerCase, TreeMap::new,

 16    }
