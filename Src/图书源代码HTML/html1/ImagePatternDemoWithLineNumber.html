 14  public class ImagePatternDemo extends Application {
 15    private static final String imageURL =
 16      "http://www.cs.armstrong.edu/liang/common/image/ohioMap.gif";
 17  
 18    @Override // Override the start method in the Application class
 19    public void start(Stage primaryStage) {
 20      Pane pane = new Pane();
 21  
 22      Rectangle rectangle1 = new Rectangle(10, 10, 50, 50);
 23      Rectangle rectangle2 = new Rectangle(70, 10, 50, 50);
 24      Rectangle rectangle3 = new Rectangle(130, 10, 50, 50);
 25      Rectangle rectangle4 = new Rectangle(190, 10, 50, 50);
 26  
 27      Image image = new Image(imageURL);

 30      rectangle3.setFill(new ImagePattern(image, 0, 0, 50, 50, false));
 31      rectangle4.setFill(new ImagePattern(image, 0, 0, 25, 25, false));
 32      
 33      pane.getChildren().addAll(rectangle1, rectangle2, 
 34        rectangle3, rectangle4);
 35      Scene scene = new Scene(pane, 300, 250);           
 36      primaryStage.setTitle("ImagePatternDemo"); // Set the window title
 37      primaryStage.setScene(scene); // Place the scene in the window
 38      primaryStage.show(); // Display the window
 39    }
